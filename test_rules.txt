# Sample Suricata/Snort rules for testing
alert tcp any any -> any 80 (msg:"HTTP Traffic Detected"; sid:1000001; rev:1;)
alert tcp any any -> any 443 (msg:"HTTPS Traffic Detected"; sid:1000002; rev:1;)
alert tcp $EXTERNAL_NET any -> $HOME_NET 22 (msg:"SSH Connection Attempt"; sid:1000003; rev:2;)
drop tcp any any -> any 3389 (msg:"RDP Connection Blocked"; sid:1000004; rev:1;)
alert udp any any -> any 53 (msg:"DNS Query Detected"; sid:1000005; rev:1;)
